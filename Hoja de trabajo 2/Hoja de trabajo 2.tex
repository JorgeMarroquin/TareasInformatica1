\section{Ejercicio \#1}
\begin{flushleft}
\textbf{Demostrar:}
\end{flushleft}
\[
        \forall\ n.\ n^3\geq n^2
\]
\begin{flushleft}
\textbf{Caso base:}
\begin{enumerate}
\item{$0^3\geq 0^2$}
\item {$0\geq 0$}
\end{enumerate}

\textbf{Caso inductivo:}
\begin{enumerate}
\item{$(n + 1)^3\geq (n + 1)^2$}
\item{$(n + 1)\geq (n + 1)^2 / (n + 1)^2$}
\item{Un número partido por él mismo es 1: \\$(n + 1)\geq 1$}
\item{$n \geq 1 - 1$}
\item{$n \geq 0$}
\end{enumerate}
\end{flushleft}

\section{Ejercicio \#2}
\begin{flushleft}
\textbf{Demostrar:}
\end{flushleft}
\[
        \forall\ n.\ (1+x)^n\geq nx
\]
\begin{flushleft}
\textbf{Caso base:}
\begin{enumerate}
\item{$(1+0)^0\geq 0(0)$}
\item {Todo número elevado a la 0 es 1:\\ $ 1 \geq 0$}
\end{enumerate}

\textbf{Caso inductivo:}
\begin{enumerate}
\item{$(1+x)^{(n + 1)}\geq (n + 1)x + 1$}
\item{$(1+x)^{(n + 1)}\geq nx + x + 1$}
\item{$(1+x).(1+x)^n\geq nx + x + 1$}

\end{enumerate}
\end{flushleft}
